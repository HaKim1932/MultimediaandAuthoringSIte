Tools & Technologies

Production Stack

The following tools and concepts were used in the production of the 3D solar system rendering project.

🎛️
3D Modeling Software
Used to create spherical meshes for each celestial body, set orbital paths, and animate planetary rotations.
💡
Lighting & Shading
Point lights simulate the Sun's radiance. Ambient and directional lights create depth and realistic shadow casting.
🖼️
Texture Mapping
High-resolution planetary surface maps (NASA textures) were UV-mapped onto spherical geometry for photorealism.
📷
Camera & Composition
Animated camera paths orbit the scene to showcase each planet, using depth-of-field to focus attention.
⚙️
Rendering Engine
A ray-tracing or rasterization engine converts 3D geometry to 2D pixels with realistic light bounce.
🎬
Video Compositing
Rendered frames were composited into a final video, with color grading applied for a cinematic space atmosphere.

for obj in list(bpy.data.objects):

for block_collection in (
    bpy.data.meshes, bpy.data.materials, bpy.data.textures,

# Schedule: Sun(108f) + 9 planets(108f each) + 9 travels(18f)
#         + Rocket montage(140f) + Grand zoom-out(117f) = 1500

FPS            = 25
TOTAL_FRAMES   = 1500
FOCUS_F        = 108   # ~4.3s per body
TRAVEL_F       = 18    # 0.72s between bodies
ASTEROID_TRAVEL_F = 60 # longer travel between Mars & Jupiter so the belt gets its own flyby moment
ROCKET_F       = 140   # 5.6s rocket montage (was 3s — too quick to register)
ZOOM_F         = 117   # ~4.7s grand pull-back (trimmed to give rockets more time)

SPEED_MULT     = 4.5
MOON_SPEED_SCALE = 0.15   # moons were orbiting way too fast (100+ laps in 60s);
                          # this scales their speed down independently of SPEED_MULT.
                          # Lower this further (e.g. 0.08) to slow them even more.

# ── Sun lighting — dimmed so planet textures stay visible ──────────
# Point light: was 55000, now 8000 (softer falloff, no blown-out look)
# Fill light on camera: was 18000, now 4000
# Sun emission: was 1.2, now 0.9 (glows but not blinding)

FILL_LIGHT_POWER      = 1200

def apply_texture(node, filename):
    filepath = os.path.join(TEXTURE_FOLDER, filename)
    if not os.path.exists(filepath):
        return
    img = bpy.data.images.get(filename)
    if img:
        img.filepath     = filepath
        img.filepath_raw = filepath
        try: img.reload()
        except Exception: pass
    else:
        img = bpy.data.images.load(filepath)
    node.image = img

def _spawn_moon(full_name, orbit_anchor, planet_dist_val,
                moon_dist, moon_radius, moon_speed, parent_orb_speed, color):
    # Moon orbit anchor (child of planet's orbit anchor, offset to planet pos)
    bpy.ops.object.empty_add(type='PLAIN_AXES', location=(0, 0, 0))
    m_anchor          = bpy.context.active_object
    m_anchor.name     = f"{full_name}_Anchor"
    m_anchor.parent   = orbit_anchor
    m_anchor.location = (planet_dist_val, 0, 0)

    start_rot = random.uniform(0, 2 * math.pi)
    m_anchor.rotation_euler[2] = start_rot
    m_anchor.keyframe_insert(data_path="rotation_euler", index=2, frame=1)
    m_anchor.rotation_euler[2] = start_rot + math.radians(
        moon_speed * SPEED_MULT * MOON_SPEED_SCALE * (TOTAL_FRAMES / 1000.0) * 360
    )
    m_anchor.keyframe_insert(data_path="rotation_euler", index=2, frame=TOTAL_FRAMES)

    # Moon sphere
    bpy.ops.mesh.primitive_uv_sphere_add(radius=max(0.05, moon_radius),
                                          segments=16, ring_count=8)
    bpy.ops.object.shade_smooth()
    moon_obj          = bpy.context.active_object
    moon_obj.name     = full_name
    moon_obj.parent   = m_anchor
    moon_obj.location = (moon_dist, 0, random.uniform(-0.08, 0.08))
    moon_obj.visible_shadow = False

    # Material — moon.png texture on every moon, tinted per-moon so we
    # keep some color variety instead of every moon looking identical.
    mm   = bpy.data.materials.new(name=f"{full_name}_Mat")
    mm.use_nodes = True
    mb   = mm.node_tree.nodes.get("Principled BSDF")
    mb.inputs['Roughness'].default_value  = 0.85
    me   = mm.node_tree.nodes.new('ShaderNodeEmission')
    mx   = mm.node_tree.nodes.new('ShaderNodeMixShader')
    mo   = mm.node_tree.nodes.get("Material Output") or mm.node_tree.nodes.new('ShaderNodeOutputMaterial')

    moon_tex_path = os.path.join(TEXTURE_FOLDER, "moon.png")
    if os.path.exists(moon_tex_path):
        m_tex = mm.node_tree.nodes.new('ShaderNodeTexImage')
        apply_texture(m_tex, "moon.png")
        tint  = mm.node_tree.nodes.new('ShaderNodeMixRGB')
        tint.blend_type = 'MULTIPLY'
        tint.inputs['Fac'].default_value    = 0.85
        tint.inputs['Color2'].default_value = color
        mm.node_tree.links.new(m_tex.outputs['Color'], tint.inputs['Color1'])
        mm.node_tree.links.new(tint.outputs['Color'],  mb.inputs['Base Color'])
        mm.node_tree.links.new(tint.outputs['Color'],  me.inputs['Color'])
    else:
        mb.inputs['Base Color'].default_value = color
        me.inputs['Color'].default_value      = color

    me.inputs['Strength'].default_value = 0.4
    mx.inputs['Fac'].default_value      = 0.3
    mm.node_tree.links.new(mb.outputs['BSDF'],     mx.inputs[1])
    mm.node_tree.links.new(me.outputs['Emission'], mx.inputs[2])
    mm.node_tree.links.new(mx.outputs['Shader'],   mo.inputs['Surface'])
    moon_obj.data.materials.append(mm)

    planet.rotation_euler[0] = math.radians(-tilt)
    planet.rotation_euler[2] = 0
    planet.keyframe_insert(data_path="rotation_euler", index=2, frame=1)
    planet.rotation_euler[2] = math.radians(
        360 * rot_speed * SPEED_MULT * (TOTAL_FRAMES / 1000.0)
    )
    planet.keyframe_insert(data_path="rotation_euler", index=2, frame=TOTAL_FRAMES)

    # Planet material
    mat   = bpy.data.materials.new(name=f"{name}_Mat")
    mat.use_nodes = True
    p_bsdf = mat.node_tree.nodes.get("Principled BSDF")
    p_tex  = mat.node_tree.nodes.new('ShaderNodeTexImage')
    p_uv   = mat.node_tree.nodes.new('ShaderNodeTexCoord')
    apply_texture(p_tex, tex_file)
    mat.node_tree.links.new(p_uv.outputs['UV'],      p_tex.inputs['Vector'])
    mat.node_tree.links.new(p_tex.outputs['Color'],  p_bsdf.inputs['Base Color'])
    p_bsdf.inputs['Roughness'].default_value = 0.55
    planet.data.materials.append(mat)

    # Saturn ring
    if name == "Saturn":
        bpy.ops.mesh.primitive_torus_add(
            major_radius=(size * 1.3 + size * 2.2) / 2,
            minor_radius=(size * 2.2 - size * 1.3) / 2,
            major_segments=64
        )
        ring          = bpy.context.active_object
        ring.name     = "Saturn_Ring"
        ring.parent   = planet
        ring.scale[2] = 0.001
        r_mat         = bpy.data.materials.new(name="Saturn_Ring_Mat")
        r_mat.use_nodes    = True
        r_mat.blend_method = 'HASHED'
        r_tex  = r_mat.node_tree.nodes.new('ShaderNodeTexImage')
        r_bsdf = r_mat.node_tree.nodes.get("Principled BSDF")

def planet_pos_at(name, frame):
    """Math-only world XY — matches orbit anchor keyframe math."""
    idx       = [p[0] for p in planets_data].index(name)
    dist      = planet_dist[name]
    orb_speed = planet_orb_sp[name]
    s_ang     = (idx + 1) * golden_angle
    e_ang     = s_ang + math.radians(
        360 * orb_speed * SPEED_MULT * (TOTAL_FRAMES / 1000.0)
    )
    t   = (frame - 1) / max(1, TOTAL_FRAMES - 1)
    ang = s_ang + t * (e_ang - s_ang)
    return (dist * math.cos(ang), dist * math.sin(ang), 0.0)


SUN_FACING_PLANETS = {"Jupiter", "Saturn", "Uranus", "Neptune"}

def cam_locked_offset(planet_name, frame=None):
    """Fixed offset — camera tracks planet with zero zoom."""
    r    = planet_size[planet_name]
    dist = max(r * 6.5, 4.0)
    up   = max(r * 2.5, 1.6)

    if planet_name in SUN_FACING_PLANETS and frame is not None:
        # Now that the Sun is dimmer, these far planets need the camera on
        # their actually-lit side or their face just looks like a dark blob.
        # Compute the real direction from the planet toward the Sun (origin)
        # at this point in its orbit, then bias the camera a bit off dead-
        # center so it's a nice three-quarter lit view instead of flat/frontal.
        px, py, _ = planet_pos_at(planet_name, frame)
        sun_dir_angle = math.atan2(-py, -px)
        a = sun_dir_angle + math.radians(30)
    else:
        angle_offsets = {
            "Mercury": math.radians(145),
            "Venus":   math.radians(150),
            "Earth":   math.radians(145),
            "Mars":    math.radians(140),
            "Pluto":   math.radians(148),
        }
        a = angle_offsets.get(planet_name, math.radians(140))
    return (math.cos(a) * dist, math.sin(a) * dist, up)

class TrailTracker:
    def __init__(self, target, name, material):
        self.target       = target
        self.trail_length = 100
        cd                = bpy.data.curves.new(name=f"{name}_C", type='CURVE')
        cd.dimensions     = '3D'

class ConstantVelocityDriver:
    def __init__(self, target, waypoints, speed):
        self.target    = target
        self.waypoints = [Vector(w) for w in waypoints]
        self.speed     = speed
        self.segments  = []
        total = 0.0
        for i in range(len(self.waypoints) - 1):
            a, b = self.waypoints[i], self.waypoints[i + 1]
            l    = (b - a).length
            self.segments.append((a, b, l)); total += l
        a, b = self.waypoints[-1], self.waypoints[0]
        l    = (b - a).length
        self.segments.append((a, b, l))
        self.total_length = total + l

def spawn_ship_on_flightpath(name, pts, mat):
    ship = build_spaceship(name)
    cd   = bpy.data.curves.new(name=f"{name}_P", type='CURVE')
    sp   = cd.splines.new('BEZIER')
    sp.use_cyclic_u = True
    sp.bezier_points.add(len(pts) - 1)
    for i, wp in enumerate(pts):
        sp.bezier_points[i].co                = wp
        sp.bezier_points[i].handle_left_type  = 'AUTO'
        sp.bezier_points[i].handle_right_type = 'AUTO'
    path_obj = bpy.data.objects.new(f"{name}_P", cd)
    bpy.context.collection.objects.link(path_obj)
    dg    = bpy.context.evaluated_depsgraph_get()
    ev    = path_obj.evaluated_get(dg)
    md    = bpy.data.meshes.new_from_object(ev)
    wpts  = [v.co.copy() for v in md.vertices]
    bpy.data.meshes.remove(md)
    tl    = sum((wpts[(i+1)%len(wpts)] - wpts[i]).length for i in range(len(wpts)))
    spd   = tl / TOTAL_FRAMES
    drv   = ConstantVelocityDriver(target=ship, waypoints=wpts, speed=spd)
    _motion_drivers.append(drv)
    _trail_trackers.append(TrailTracker(ship, f"{name}_Trail", mat))

def bake_all_space_physics():
    motion_paths = {}
    for d in _motion_drivers:
        positions = []
        for frame in range(1, TOTAL_FRAMES + 1):
            dist = ((frame - 1) * d.speed) % d.total_length
            pos  = d.waypoints[-1]
            for (a, b, length) in d.segments:
                if dist <= length:
                    pos = a.lerp(b, dist / length if length > 0 else 0)
                    break
                dist -= length
            positions.append(pos.copy())
        motion_paths[id(d)] = positions
        ROCKET_POSITIONS[d.target.name] = positions
        d.target.rotation_mode = 'QUATERNION'
        for frame in range(1, TOTAL_FRAMES + 1):
            idx = frame - 1
            d.target.location = positions[idx]
            fwd = ((positions[idx+1] - positions[idx])
                   if idx+1 < len(positions)
                   else (positions[idx] - positions[idx-1]))
            fwd.normalize()
            d.target.rotation_quaternion = fwd.to_track_quat('X', 'Z')
            d.target.keyframe_insert(data_path="location",            frame=frame)
            d.target.keyframe_insert(data_path="rotation_quaternion", frame=frame)

    for tracker in _trail_trackers:
        spline = tracker.spline
        n      = tracker.trail_length
        if len(spline.points) < n:
            spline.points.add(n - len(spline.points))
        for i in range(n):
            spline.points[i].radius = (i / max(1, n - 1)) ** 1.5
        t_path = motion_paths[
            next(id(d) for d in _motion_drivers if d.target == tracker.target)
        ]
        for frame in range(1, TOTAL_FRAMES + 1):
            for i in range(n):
                pos = t_path[max(0, (frame-1) - (n-1-i))]
                spline.points[i].co = (pos.x, pos.y, pos.z, 1.0)
                spline.points[i].keyframe_insert(data_path="co", frame=frame)
